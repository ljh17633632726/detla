三角洲平台 - 部署文档

本系统交付物为三部分：打包后的微信小程序、Java 后端 jar、PC 管理后台静态包。本文档说明如何在服务器上部署这些内容。

交付物说明部署方式
微信小程序已打包的小程序代码目录用微信开发者工具上传，不部署在服务器
Java 后端delta-admin.jar服务器上运行 jar
PC 管理后台admin-ui 的 dist 目录服务器上由 Nginx 提供静态资源
---

一、环境要求

  • JDK：17+
  • MySQL：8.0+
  • Redis：可用即可
  • Nginx：用于反向代理和静态资源
---

二、服务器目录与文件放置

服务器请安装宝塔面板，系统选择 Ubuntu 20.04.4（20 版本即可）

建议在服务器上建一个统一目录（如 /www/wwwroot/），按下面方式放置。

重要：jar 包先不要上传，需在下面第四步配置完成后再上传！

/www/wwwroot                      # 根目录
├── delta-admin.jar  # 放置提供的 Java 后端 jar
│
├── dist/           #新建一个目录，用于放置 PC 端 dist 静态资源
│   └── XXX                  # 放置提供的 PC 端 dist 整个目录里的所有文件

三、Nginx 配置（API + 文件 + PC 管理后台）

1. 将下方配置文件中的 XXXXXXX.com 全部替换为您的实际域名


2. 将 ssl_certificate 和 ssl_certificate_key 中的 XXXXXXX 替换为您的 SSL 证书文件名


3. 其他无需修改，直接复制粘贴即可


4. 配置完成后务必重启 Nginx


5. 验证是否成功：直接访问域名 https://您的域名，能打开页面即表示配置成功

user www www;
worker_processes auto;
error_log /www/wwwlogs/nginx_error.log crit;
pid /www/server/nginx/logs/nginx.pid;
worker_rlimit_nofile 51200;

stream {
    log_format tcp_format '$time_local|$remote_addr|$protocol|$status|$bytes_sent|$bytes_received|$session_time|$upstream_addr|$upstream_bytes_sent|$upstream_bytes_received|$upstream_connect_time';
    access_log /www/wwwlogs/tcp-access.log tcp_format;
    error_log /www/wwwlogs/tcp-error.log;
    include /www/server/panel/vhost/nginx/tcp/*.conf;
}


events {
    use epoll;
    worker_connections 51200;
    multi_accept on;
}


http {


include mime.types;
    include proxy.conf;
        lua_package_path "/www/server/nginx/lib/lua/?.lua;;";


default_type application/octet-stream;


sendfile on;
    tcp_nopush on;
    tcp_nodelay on;


keepalive_timeout 60;
    client_max_body_size 50m;


gzip on;
    gzip_min_length 1k;
    gzip_buffers 4 16k;
    gzip_http_version 1.1;
    gzip_comp_level 5;
    gzip_types text/plain application/javascript text/css application/json application/xml image/png image/jpeg image/svg+xml;
    gzip_vary on;


server_tokens off;


# ===============================
# phpMyAdmin
# ===============================

    listen 888;
    server_name phpmyadmin;

        allow 127.0.0.1;
        allow ::1;
        deny all;
    index index.php index.html;


include enable-php.conf;


location ~ /\. {
        deny all;
    }


access_log /www/wwwlogs/access.log;
}


# ===============================
# HTTP 自动跳转 HTTPS
# ===============================

listen 80;

}


# ===============================
# 主站 HTTPS
# ===============================
server {


listen 443 ssl;
    http2 on;


server_name XXXXXXX.com www.XXXXXXX.com;


root /www/wwwroot/dist;
    index index.html;


ssl_certificate /www/ssl/XXXXXXX.pem;
    ssl_certificate_key /www/ssl/XXXXXXX.key;


ssl_protocols TLSv1.2 TLSv1.3;


# =========================
    # Vue 前端
    # =========================
    location / {
        try_files $uri $uri/ /index.html;
    }


# =========================
    # API 代理（保留 /api）
    # =========================
    location /api/ {


proxy_pass http://127.0.0.1:8080/;


proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;

# =========================
    # WebSocket
    # =========================
location /ws/chat {


proxy_pass http://127.0.0.1:8080;


proxy_http_version 1.1;

    proxy_set_header Connection "upgrade";

    proxy_set_header X-Real-IP $remote_addr;


proxy_read_timeout 86400;


}
    location /file/ {
        alias /www/data/upload/;
        expires 30d;
        access_log off;
    }


# 静态资源缓存
    location ~* \.(js|css)$ {
        expires 7d;
    }
}


include /www/server/panel/vhost/nginx/*.conf;


}
